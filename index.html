const VideoPlayer = ({ title, thumbnailClass, videoUrl, isMain }) => {

  const [isPlaying, setIsPlaying] = useState(false);


  // Reset play state when video URL changes

  useEffect(() => {

    setIsPlaying(false);

  }, [videoUrl]);


  return (

    <div className="relative w-full aspect-video rounded-lg overflow-hidden border border-slate-700 shadow-xl group bg-black/80 backdrop-blur-sm transition-all duration-500">

      {isPlaying && videoUrl ? (

        <iframe 

          className="w-full h-full animate-fade-in"

          src={`${videoUrl}?autoplay=1`} 

          title={title}

          frameBorder="0" 

          allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" 

          allowFullScreen

        ></iframe>

      ) : (

        <div className={`w-full h-full ${thumbnailClass} flex items-center justify-center transition-all duration-500 group-hover:opacity-90 relative`}>

          <div className="absolute inset-0 bg-black bg-opacity-40 group-hover:bg-opacity-30 transition-all" />

          

          <div className="z-10 text-center px-4">

             <h3 className="text-white font-bold text-xl mb-2 drop-shadow-md">

               {isMain ? `${title} Main Trailer` : title}

             </h3>

             {videoUrl ? (

               <button 

                  onClick={() => setIsPlaying(true)}

                  className="bg-white/10 hover:bg-indigo-500 text-white rounded-full p-4 backdrop-blur-sm border border-white/20 transition-all transform hover:scale-110"

               >

                 <Play fill="currentColor" size={32} />

               </button>

             ) : (

               <div className="flex flex-col items-center">

                 <p className="text-slate-400 text-sm bg-black/50 px-3 py-1 rounded border border-white/10">No video link available</p>

               </div>

             )}

          </div>

        </div>

      )}

    </div>

  );

};
